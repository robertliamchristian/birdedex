function fetchSuggestions() { var input = document.getElementById('bird-input'); var query = input.value; } if (query.length < 2) { document.getElementById('suggestions').style.display = 'none'; return; } fetch('/suggest_birds?query=' + query) .then(response => response.json()) .then(suggestions => { var suggestionsDiv = document.getElementById('suggestions'); suggestionsDiv.innerHTML = ''; suggestions.forEach(function(suggestion) { var div = document.createElement('div'); div.innerHTML = suggestion; div.onclick = function() { input.value = suggestion; suggestionsDiv.style.display = 'none'; }; suggestionsDiv.appendChild(div); }); if (suggestions.length > 0) { suggestionsDiv.style.display = 'block'; } else { suggestionsDiv.style.display = 'none'; } }) .catch(error => console.error('Error fetching suggestions:', error)); } window.onclick = function(event) { if (!event.target.matches('#bird-input')) { document.getElementById('suggestions').style.display = 'none'; } };
